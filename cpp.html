std::vector Documentation

1. Constructors

Default Constructor

Initializes an empty vector.

[Click to view example] 

#include <vector>
#include <iostream>

int main() {
    std::vector v;
    std::cout << "Size: " << v.size() << std::endl;
    return 0;
}

Fill Constructor

Initializes a vector with specified size and value.

[Click to view example] 

#include <vector>
#include <iostream>

int main() {
    std::vector v(5, 10);
    for (int x : v) {
        std::cout << x << " ";
    }
    return 0;
}

2. Element Access

Access Element (operator[])

Accesses an element by index without bounds checking.

[Click to view example] 

#include <vector>
#include <iostream>

int main() {
    std::vector v = {1, 2, 3, 4};
    std::cout << v[2] << std::endl; // Output: 3
    return 0;
}

Access Element (at)

Accesses an element by index with bounds checking.

[Click to view example] 

#include <vector>
#include <iostream>

int main() {
    std::vector v = {1, 2, 3, 4};
    try {
        std::cout << v.at(2) << std::endl; // Output: 3
        std::cout << v.at(10) << std::endl; // Throws an exception
    } catch (std::out_of_range& e) {
        std::cerr << e.what() << std::endl;
    }
    return 0;
}
